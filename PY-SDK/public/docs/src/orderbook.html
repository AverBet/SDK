219    @staticmethod
220    def __get_L2(slab: Slab, depth: int, decimals: int, increasing: bool):
221        """Get the Level 2 market information."""
222        # The first element of the inner list is price, the second is quantity.
223        levels: list[list[int]] = []
224        for node in slab.items(descending=not increasing):
225            price = Orderbook.__get_price_from_slab(node, decimals)
226            if len(levels) > 0 and levels[len(levels) - 1][0] == price:
227                levels[len(levels) - 1][1] += node.base_quantity
228            elif len(levels) == depth:
229                break
230            else:
231                levels.append([price, node.base_quantity])
232        return [
233            Price(
234                price=price_lots, 
235                size=size_lots
236            )
237            for price_lots, size_lots in levels
238        ]

240    @staticmethod
241    def __get_L1(slab: Slab, decimals: int, increasing: bool, is_inverted: bool):
242        """Get the Level 1 market information."""
243        # The first element of the inner list is price, the second is quantity.
244        orders: list[SlabOrder] = []
245        for node in slab.items(descending = not increasing):
246            orders += [SlabOrder(
247                id = node.key,
248                price = (10**decimals) - Orderbook.__get_price_from_slab(node, decimals) if is_inverted else Orderbook.__get_price_from_slab(node, decimals),
249                price_ui = 1 - Orderbook.__get_price_from_slab(node, decimals) * (10**-decimals) if is_inverted else Orderbook.__get_price_from_slab(node, decimals) * (10**-decimals),
250                base_quantity = node.base_quantity,
251                base_quantity_ui = node.base_quantity * (10**-decimals),
252                user_market = node.user_market,
253                fee_tier = node.fee_tier,
254            )]
255
256        return orders

475    def __estimate_fill_for_qty(self, qty: int, side: Side, quote: bool, ui_amount: bool):
476        """
477        _summary_
478
479        Args:
480            qty (int): Quanity
481            side (Side): Side object (bid or ask)
482            quote (bool): Quote quantity if true. Base quantity if false.
483            ui_amount (bool): Converts prices based on decimal precision if true.
484
485        Returns:
486            dict[str, float]: Dictionary containing `avg_price`, `worst_price`, `filled`
487        """
488        if(side == Side.BUY):
489            prices = self.get_bids_l2(100, ui_amount)
490        elif(side == Side.SELL):
491            prices = self.get_asks_l2(100, ui_amount)
492        
493        if(quote):
494            accumulator = lambda p: p.size
495        else:
496            accumulator = lambda p: p.size * p.price
497        
498        new_prices: list[Price] = []
499        cumulative_qty = 0
500        for price in prices:
501            remaining_qty = qty - cumulative_qty
502            if(remaining_qty <= accumulator(price)):
503                cumulative_qty += remaining_qty
504                new_size = remaining_qty if quote else remaining_qty/price.price
505                new_prices.append(Price(price=price.price, size=new_size))
506                break
507            else:
508                cumulative_qty += accumulator(price)
509                new_prices.append(price)
510        
511        return {
512            'avg_price': Orderbook.weighted_average(
513                nums=[p.price for p in new_prices],
514                weights=[p.size for p in new_prices]
515            ),
516            'worst_price': new_prices[-1].price,
517            'filled': cumulative_qty
518        }

521    def weighted_average(nums, weights):
522        """
523        Calculates weighted average
524
525        Args:
526            nums (list[float]): List of values
527            weights (list[float]): List of weights
528
529        Returns:
530            float: Weighted average
531        """
532        sum = 0
533        weight_sum = 0
534
535        assert len(nums) == len(weights), 'Number of weights and nums do not correspond'
536
537        for i, num in enumerate(nums):
538            weight = weights[i]
539            sum += num * weight
540            weight_sum += weight
541        
542        return sum / weight_sum

220    @staticmethod
221    def __get_L2(slab: Slab, depth: int, decimals: int, increasing: bool):
222        """Get the Level 2 market information."""
223        # The first element of the inner list is price, the second is quantity.
224        levels: list[list[int]] = []
225        for node in slab.items(descending=not increasing):
226            price = Orderbook.__get_price_from_slab(node, decimals)
227            if len(levels) > 0 and levels[len(levels) - 1][0] == price:
228                levels[len(levels) - 1][1] += node.base_quantity
229            elif len(levels) == depth:
230                break
231            else:
232                levels.append([price, node.base_quantity])
233        return [
234            Price(
235                price=price_lots, 
236                size=size_lots
237            )
238            for price_lots, size_lots in levels
239        ]

243        """Get the Level 1 market information."""
244        # The first element of the inner list is price, the second is quantity.
245        orders: list[SlabOrder] = []
246        for node in slab.items(descending = not increasing):
247            orders += [SlabOrder(
248                id = node.key,

250                price_ui = 1 - Orderbook.__get_price_from_slab(node, decimals) * (10**-decimals) if is_inverted else Orderbook.__get_price_from_slab(node, decimals) * (10**-decimals),
251                base_quantity = node.base_quantity,
252                base_quantity_ui = node.base_quantity * (10**-decimals),
253                user_market = node.user_market,
254                fee_tier = node.fee_tier,
255            )]

476    def __estimate_fill_for_qty(self, qty: int, side: Side, quote: bool, ui_amount: bool):
477        """
478        _summary_
479
480        Args:
481            qty (int): Quanity
482            side (Side): Side object (bid or ask)
483            quote (bool): Quote quantity if true. Base quantity if false.
484            ui_amount (bool): Converts prices based on decimal precision if true.
485
486        Returns:
487            dict[str, float]: Dictionary containing `avg_price`, `worst_price`, `filled`
488        """
489        if(side == Side.BUY):
490            prices = self.get_bids_l2(100, ui_amount)
491        elif(side == Side.SELL):
492            prices = self.get_asks_l2(100, ui_amount)
493        
494        if(quote):
495            accumulator = lambda p: p.size
496        else:
497            accumulator = lambda p: p.size * p.price
498        
499        new_prices: list[Price] = []
500        cumulative_qty = 0
501        for price in prices:
502            remaining_qty = qty - cumulative_qty
503            if(remaining_qty <= accumulator(price)):
504                cumulative_qty += remaining_qty
505                new_size = remaining_qty if quote else remaining_qty/price.price
506                new_prices.append(Price(price=price.price, size=new_size))
507                break
508            else:
509                cumulative_qty += accumulator(price)
510                new_prices.append(price)
511        
512        return {
513            'avg_price': Orderbook.weighted_average(
514                nums=[p.price for p in new_prices],
515                weights=[p.size for p in new_prices]
516            ),
517            'worst_price': new_prices[-1].price,
518            'filled': cumulative_qty
519        }

523        """
524        Calculates weighted average
525
526        Args:
527            nums (list[float]): List of values
528            weights (list[float]): List of weights
529
530        Returns:
531            float: Weighted average
532        """
533        sum = 0
534        weight_sum = 0
535
536        assert len(nums) == len(weights), 'Number of weights and nums do not correspond'
537
538        for i, num in enumerate(nums):
539            weight = weights[i]
540            sum += num * weight
541            weight_sum += weight
542        
543        return sum / weight_sum

521    @staticmethod
522    def weighted_average(nums, weights):
523        """
524        Calculates weighted average
525
526        Args:
527            nums (list[float]): List of values
528            weights (list[float]): List of weights
529
530        Returns:
531            float: Weighted average
532        """
533        sum = 0
534        weight_sum = 0
535
536        assert len(nums) == len(weights), 'Number of weights and nums do not correspond'
537
538        for i, num in enumerate(nums):
539            weight = weights[i]
540            sum += num * weight
541            weight_sum += weight
542        
543        return sum / weight_sum
