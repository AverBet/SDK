 348        """
 349        Attempts to load UserMarket object and creates one if not found
 350
 351        Args:
 352            client (AverClient): AverClient object
 353            owner (Keypair): Owner of UserMarket account
 354            market (AverMarket): Corresponding AverMarket object
 355            send_options (TxOpts, optional): Options to specify when broadcasting a transaction. Defaults to None.

 357            host (PublicKey, optional): Host account public key. Defaults to AVER_HOST_ACCOUNT.

 360            discount_token (PublicKey, optional): _description_. Defaults to SYS_PROGRAM_ID.

 364            UserMarket: UserMarket object
 365        """

 367        if(number_of_orders is None):
 368            number_of_orders = market.market_state.number_of_outcomes * 5
 369        
 370        user_market_pubkey = UserMarket.derive_pubkey_and_bump(owner.public_key, market.market_pubkey, host, program_id)[0]

 372            uma = await UserMarket.load(client, market, owner.public_key, host, program_id)

 375            uhl = await UserHostLifetime.get_or_create_user_host_lifetime(
 376                client,
 377                owner,
 378                send_options,
 379                quote_token_mint,
 380                host,
 381                referrer,

 383                program_id
 384            )
 385
 386            sig = await UserMarket.create_user_market_account(
 387                client,
 388                market,
 389                owner, 
 390                send_options,
 391                host,
 392                number_of_orders,
 393                program_id,
 394            )
 395
 396            await client.provider.connection.confirm_transaction(
 397                sig['result'],
 398                commitment=Finalized
 399            )
 400
 401            return await UserMarket.load(
 402                client, 
 403                market,  
 404                owner.public_key,
 405                host,
 406                program_id)

 445
 446        Args:
 447            outcome_position (int): ID of outcome
 448            side (Side): Side object (bid or ask)
 449            limit_price (float): Limit price
 450            size (float): Size
 451            size_format (SizeFormat): SizeFormat object (Stake or Payout)
 452            user_quote_token_ata (PublicKey): Quote token ATA public key (holds funds for this user)
 453            order_type (OrderType, optional): OrderType object. Defaults to OrderType.LIMIT.

 476

 480                "size": max_base_qty,
 481                "size_format": size_format,
 482                "side": side,
 483                "order_type": order_type,
 484                "self_trade_behavior": self_trade_behavior,

 570    def make_cancel_order_instruction(
 571        self,
 572        order_id: int,
 573        outcome_position: int,
 574        active_pre_flight_check: bool = False,
 575    ):
 576        """
 577        Creates instruction for to cancel order.
 578
 579        Returns TransactionInstruction object only. Does not send transaction.
 580
 581        Args:
 582            order_id (int): ID of order to cancel
 583            outcome_position (int): ID of outcome
 584            active_pre_flight_check (bool, optional): Clientside check if order will success or fail. Defaults to False.
 585
 586        Raises:
 587            Exception: Cannot cancel orders on closed market
 588            Exception: Insufficient lamport balance
 589            Exception: Cannot cancel orders in current market status
 590            Exception: Order ID does not exist in list of open orders
 591
 592        Returns:
 593            TransactionInstruction: TransactionInstruction object
 594        """
 595        if(self.market.orderbooks is None):
 596            raise Exception('Cannot cancel orders on closed market')
 597
 598        if(active_pre_flight_check):
 599            if(self.user_balance_state.lamport_balance < 5000):
 600                raise Exception('Insufficient lamport balance')
 601            
 602            if(not can_cancel_order_in_market(self.market.market_state.market_status)):
 603                raise Exception('Cannot cancel orders in current market status')
 604            
 605            if(not str(order_id) in [str(o.order_id) for o in self.user_market_state.orders]):
 606                raise Exception('Order ID does not exist in list of open orders')
 607

 689        """

 702            if(not outcome_ids_to_cancel.__contains__(i)):
 703                continue
 704            remaining_accounts += [AccountMeta(
 705                pubkey=accounts.orderbook,
 706                is_signer=False,

 720                pubkey=accounts.asks,
 721                is_signer=False,
 722                is_writable=True,
 723            )]
 724        
 725        chunk_size = 5
 726        chunked_outcome_ids = chunk(outcome_ids_to_cancel, chunk_size)
 727        chunked_remaining_accounts = chunk(remaining_accounts, chunk_size * 4)

1052        outcome_index: int,
1053        side: Side,
1054        limit_price: float,
1055        size: float,
1056        size_format: SizeFormat
1057    ):

1059        Performs clientside checks prior to placing an order
1060
1061        Args:
1062            outcome_index (int): Outcome ID
1063            side (Side): Side
1064            limit_price (float): Limit price
1065            size (float): Size
1066            size_format (SizeFormat): SizeFormat object (state or payout)

1071            Exception: Max number of orders recieved

1081        current_balance = self.calculate_tokens_available_to_sell(outcome_index, limit_price) if side == Side.SELL else self.calculate_tokens_available_to_buy(outcome_index, limit_price)

 686            Exception: Cannot cancel orders in current market status

 688        Returns:
 689            TransactionInstruction: TransactionInstruction object

 700
 701        remaining_accounts: list[AccountMeta] = []
 702        for i, accounts in enumerate(self.market.market_store_state.orderbook_accounts):
 703            if(not outcome_ids_to_cancel.__contains__(i)):
 704                continue
 705            remaining_accounts += [AccountMeta(
 706                pubkey=accounts.orderbook,
 707                is_signer=False,
 708                is_writable=True,

 721                pubkey=accounts.asks,
 722                is_signer=False,
 723                is_writable=True,
 724            )]
 725        
 726        chunk_size = 5

 728        chunked_remaining_accounts = chunk(remaining_accounts, chunk_size * 4)
